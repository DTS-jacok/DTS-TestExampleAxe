Function my_edit_set(ByVal objectname As String, ByVal value As String) As Integer

    HTMLEditBox(objectname).Attach
    HTMLEditBox.Clear
    HTMLEditBox.SetFocus
    Keyboard.Type (value)
    
    my_edit_set = 0
    
End Function

Function my_obj_exists(ByVal objectname As String) As String

    HTMLEditBox(objectname).Attach
    
    If HTMLEditBox.Exists = True Then
        my_obj_exists = "True"
    Else
        my_obj_exists = "False"
    End If
    
End Function


Function MyValidateLength(ByVal expected As String, ByVal actual As String) As Integer

    Dim res As Integer
    Dim actual_length As Integer
    
    actual_length = Len(actual)
    
    If actual_length = Val(expected) Then
        res = 0
    Else
        res = 1
    End If
    
    MyValidateLength = AxeMainAPI.StepValidate(expected, actual_length, res)

End Function
