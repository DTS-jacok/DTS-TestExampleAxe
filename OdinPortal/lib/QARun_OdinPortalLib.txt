Function my_edit_set(objectname,value):Var

    Var rc
    
    rc = EditText( "@"+objectname, "")
    Type (value)
    
    If rc = 1 Then
        rc = 0
    Else
        rc = 1
    EndIf    
    
    return rc
    
End Function


Function my_obj_exists(objectname):Var

    Var ret

    ret = IsWindow(objectname,"exists")
    
    If ret = 1 Then
        return "True"
    Else
        return "False"
    EndIf
    
End Function


Function MyValidateLength(expected,actual):Var

    Var res
    Var actual_length
    
    actual_length = Length(actual)
    
    If actual_length = Val(expected) Then
        res = 0
    Else
        res = 1
    EndIf
    
    return AxeStepValidate(expected, actual_length, res)

End Function
